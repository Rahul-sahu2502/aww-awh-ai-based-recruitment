-- 1. Create the Audit table for post_map
CREATE TABLE `tbl_audit_post_map` (
    `audit_id` INT NOT NULL AUTO_INCREMENT, -- Unique ID for each audit entry
    `original_id` INT NOT NULL, -- The original primary key of the updated row from post_map
    `old_fk_post_id` INT NOT NULL DEFAULT '0',
    `old_fk_district_id` INT NOT NULL,
    `old_created_at` TIMESTAMP NULL,
    `old_updated_at` TIMESTAMP NULL,
    `audit_timestamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, -- When this audit record was created
    `audit_action` VARCHAR(10) NOT NULL DEFAULT 'UPDATE', 
    PRIMARY KEY (`audit_id`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
;

-- 2. Create the BEFORE UPDATE Trigger for post_map

DELIMITER //

CREATE TRIGGER `trg_post_map`
BEFORE UPDATE ON `post_map`
FOR EACH ROW
BEGIN
    INSERT INTO `tbl_audit_post_map` (
        `original_id`,
        `old_fk_post_id`,
        `old_fk_district_id`,
        `old_created_at`,
        `old_updated_at`
    )
    VALUES (
        OLD.`id`,
        OLD.`fk_post_id`,
        OLD.`fk_district_id`,
        OLD.`created_at`,
        OLD.`updated_at`
    );
END //

DELIMITER ;
